算法分类

一般的，算法可分为两类：非线性时间比较类排序、线性时间非比较类排序

非线性时间比较类排序：通过比较来决定元素间的相对次序，由于其时间复杂度不能突破O(nlogn)，因此称为非线性时间比较类排序。

线性时间非比较类排序：不通过比较来决定元素间的相对次序，它可以突破基于比较排序的时间下界，以线性时间运行，因此称为线性时间非比较类排序。

冒泡排序

/*
冒泡排序
时间复杂度：o(n2)
 */
@Override
public void bubbleSort(int[] data) {
    for (int i=0;i<data.length-1;i++){  //控制排序趟数（n个数排序n-1次就行，最后一次不用排，剩下的数就是最大或最小的数）
        for (int j=0;j<data.length-1-i;j++){    //控制每趟比较的次数，每趟比较的次数逐渐减少
            if (data[j]>data[j+1]){
                int temp=data[j];
                data[j]=data[j+1];
                data[j+1]=temp;
            }
        }
    }
}

快速排序

/*
快速排序:递归分治思想（中间点为参考点，左边的值比中间值小，右边的值比中间值大，依此类推）
 */
@Override
public void quickSort(int[] data,int low,int high) {
    if (low>high){
        return;
    }
    int l=low;
    int h=high;
    int basic=data[low];    //基准参考点默认首元素
    while (l!=h){
        while (l<h && data[h]>=basic){      //从后向前比较
            h--;
        }
        while (l<h && data[l]<=basic){      //从前向后比较
            l++;
        }
        if(l<h){    //交换二者位置
            int temp=data[h];
            data[h]=data[l];
            data[l]=temp;
        }
    }
    //最终交换结束后参考值归位
    data[low]=data[l];
    data[l]=basic;
    quickSort(data,low,l-1);//继续处理左边的，这里是一个递归的过程
    quickSort(data,l+1,high);//继续处理右边的 ，这里是一个递归的过程
}

插入排序

简单插入排序

/*
插入排序:若key值前的元素比key值大，则将元素依次后移，知道比key值小时将key值插入该位置
 */
@Override
public void insertionSort(int[] data) {
    int i,j;
    for (i=1;i<data.length;i++){
        int key=data[i];        //key为待插入的值
        for (j=i-1;j>=0 && data[j]>key;j--){
            data[j+1]=data[j];
        }
        data[j+1]=key;
    }
}

简单选择排序

/*
选择排序
 */
@Override
public void selectionSort(int[] data) {
    int min=0,j=0,temp=0;
    for (int i=0;i<data.length-1;i++){
        min=i;      //标记最小值的索引
        for (j=i+1;j<data.length;j++){
            if (data[j]<data[min]){
                min=j;
            }
        }
        temp=data[i];
        data[i]=data[min];
        data[min]=temp;
    }
}
